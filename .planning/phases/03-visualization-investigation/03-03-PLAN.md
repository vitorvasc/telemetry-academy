       export function evaluateGuess(
         guessId: string,
         data: Phase2Data,
         caseId: string
       ): EvaluationResult {
         const rules = createDefaultRules(caseId);
         const rule = rules.get(guessId);
         
         if (!rule) {
           return {
             correct: false,
             explanation: 'Invalid selection. Please choose a valid option.',
           };
         }
         
         const correct = rule.evaluate(data);
         return {
           correct,
           explanation: correct
             ? rule.explainCorrect(data)
             : rule.explainIncorrect(data, guessId),
           hint: correct ? undefined : getHintForGuess(guessId, data),
         };
       }
